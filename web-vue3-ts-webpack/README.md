// 卸载旧版本

npm uninstall vue-cli -g



// 安装最新vue cli 5

npm install -g @vue/cli



// 查看版本

Vue -V

输出：@vue/cli 5.0.8



// 创建项目

vue create demo

选择：Manually select features

​           features勾选：Typescript,Router,Vuex,CSS Pre-processors这四个

​           后面的全部默认



// 测试运行

cd demo

npm run serve
